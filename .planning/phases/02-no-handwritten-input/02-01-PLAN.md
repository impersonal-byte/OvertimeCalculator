    object DurationMapper {
        const val STEP_SIZE = 30
        
        fun snapToStep(value: Float, minMinutes: Int, maxMinutes: Int, stepSize: Int = STEP_SIZE): Float {
            val stepped = ((value / stepSize).toInt() * stepSize).toFloat()
            return stepped.coerceIn(minMinutes.toFloat(), maxMinutes.toFloat())
        }
        
        fun formatDuration(minutes: Int): String {
            val hours = minutes / 60.0
            return "${hours.toInt().let { if (it == 0 && hours < 0) "-0" else it }}.${kotlin.math.abs((hours % 1) * 10).toInt()}h"
        }
    }
    
    # VisualCenterMapper: Maps signed minutes to normalized slider position with zero at center
    # Key insight: Instead of naive -480..960 mapping (zero at 34%), we use a piecewise approach:
    # - Negative range: -480 to 0 maps to 0.0 to 0.5 (50% of visual space for -8h to 0)
    # - Positive range: 0 to 960 maps to 0.5 to 1.0 (50% of visual space for 0 to +16h)
    object VisualCenterMapper {
        /**
         * Converts signed minutes to normalized slider position (0.0 to 1.0).
         * When centered=true: zero appears at 0.5 (center of slider track).
         * When centered=false: zero appears at 0.333 (naive linear mapping).
         */
        fun minutesToSlider(
            minutes: Int,
            minMinutes: Int,
            maxMinutes: Int,
            centered: Boolean = true
        ): Float {
            return if (centered) {
                val totalRange = maxMinutes - minMinutes  // 1440
                val negativeRange = -minMinutes  // 480 (absolute value of most negative)
                val zeroPosition = negativeRange.toFloat() / totalRange  // 0.333
                
                if (minutes <= 0) {
                    // Map -480..0 to 0.0..0.5
                    val negativeSpan = minutes - minMinutes  // 0 to 480
                    zeroPosition * (negativeSpan.toFloat() / negativeRange)
                } else {
                    // Map 0..960 to 0.5..1.0
                    val positiveSpan = minutes.toFloat()  // 0 to 960
                    zeroPosition + (positiveSpan / negativeRange) * (1f - zeroPosition)
                }
            } else {
                // Naive linear mapping (zero at 0.333)
                (minutes - minMinutes).toFloat() / (maxMinutes - minMinutes)
            }
        }
        
        /**
         * Converts normalized slider position (0.0 to 1.0) back to signed minutes.
         */
        fun sliderToMinutes(
            sliderPosition: Float,
            minMinutes: Int,
            maxMinutes: Int,
            centered: Boolean = true
        ): Int {
            return if (centered) {
                val totalRange = maxMinutes - minMinutes  // 1440
                val negativeRange = -minMinutes  // 480
                val zeroPosition = negativeRange.toFloat() / totalRange  // 0.333
                
                if (sliderPosition <= zeroPosition) {
                    // Map 0.0..0.5 to -480..0
                    val negativePortion = sliderPosition / zeroPosition
                    (negativePortion * negativeRange + minMinutes).toInt()
                } else {
                    // Map 0.5..1.0 to 0..960
                    val positivePortion = (sliderPosition - zeroPosition) / (1f - zeroPosition)
                    (positivePortion * negativeRange).toInt()
                }
            } else {
                // Naive linear mapping
                (sliderPosition * (maxMinutes - minMinutes) + minMinutes).toInt()
            }
        }
    }

    # Key design decisions:
    # 1. Use VisualCenterMapper to make zero appear at 50% of slider width
    # 2. Add custom drawScope for sparse major tick markers only
    # 3. Keep 30-minute internal snapping via DurationMapper
    
    @Composable
    fun CenteredDurationSlider(
        value: Float,
        onValueChange: (Float) -> Unit,
        minMinutes: Int,
        maxMinutes: Int = OvertimeEntryValidator.MAX_OVERTIME_MINUTES,
        modifier: Modifier = Modifier,
        enabled: Boolean = true,
        centeredVisual: Boolean = true,
    ) {
        // Convert signed minutes to normalized slider position (0.0 to 1.0)
        val sliderPosition = VisualCenterMapper.minutesToSlider(
            value.toInt(), minMinutes, maxMinutes, centeredVisual
        )
        
        // Calculate steps based on the normalized range
        val totalSteps = ((maxMinutes - minMinutes) / DurationMapper.STEP_SIZE) - 1
        
        Column(modifier = modifier) {
            Slider(
                value = sliderPosition,
                onValueChange = { newPosition ->
                    // Convert back to signed minutes
                    val newMinutes = VisualCenterMapper.sliderToMinutes(
                        newPosition, minMinutes, maxMinutes, centeredVisual
                    )
                    // Snap to nearest 30-minute step
                    val snapped = DurationMapper.snapToStep(
                        newMinutes.toFloat(), minMinutes, maxMinutes
                    )
                    onValueChange(snapped)
                },
                valueRange = 0f..1f,
                steps = totalSteps.coerceAtLeast(0),
                enabled = enabled,
                modifier = Modifier.testTag("duration_slider"),
                onValueChangeFinished = {
                    LocalHapticFeedback.current.performHapticFeedback(HapticFeedbackType.TextHandleMove)
                },
            )
            
            # Sparse major tick markers - NOT every 30 minutes
            # Major ticks at: -8h, -4h, 0, 4h, 8h, 12h, 16h
            MajorTickRow(
                minMinutes = minMinutes,
                maxMinutes = maxMinutes,
                centeredVisual = centeredVisual,
                modifier = Modifier.testTag("major_ticks"),
            )
            
            # Center marker at zero position
            CenterMarker(
                minMinutes = minMinutes,
                maxMinutes = maxMinutes,
                centeredVisual = centeredVisual,
                modifier = Modifier.testTag("center_marker"),
            )
        }
    }
    
    @Composable
    private fun MajorTickRow(
        minMinutes: Int,
        maxMinutes: Int,
        centeredVisual: Boolean,
        modifier: Modifier = Modifier,
    ) {
        # Sparse major ticks: every 4 hours (240 minutes)
        # For workday: -480, -240, 0, 240, 480, 720, 960
        # For non-workday: 0, 240, 480, 720, 960
        val majorTickMinutes = generateSequence(
            if (minMinutes < 0) minMinutes else 0
        ) { it + 240 }.takeWhile { it <= maxMinutes }.toList()
        
        Row(
            modifier = modifier.fillMaxWidth(),
            horizontalArrangement = Arrangement.SpaceBetween,
        ) {
            majorTickMinutes.forEach { mins ->
                val position = VisualCenterMapper.minutesToSlider(mins, minMinutes, maxMinutes, centeredVisual)
                val label = DurationMapper.formatDuration(mins)
                Text(
                    text = label,
                    style = MaterialTheme.typography.labelSmall,
                    modifier = Modifier
                        .testTag("tick_$mins")
                        # Position based on slider track position
                )
            }
        }
    }
    
    @Composable
    private fun CenterMarker(
        minMinutes: Int,
        maxMinutes: Int,
        centeredVisual: Boolean,
        modifier: Modifier = Modifier,
    ) {
        # Draw a vertical line at the zero position
        val zeroPosition = VisualCenterMapper.minutesToSlider(0, minMinutes, maxMinutes, centeredVisual)
        Box(
            modifier = modifier
                .fillMaxWidth()
                .height(4.dp)
        ) {
            # Zero marker line - positioned at 50% when centered
            Divider(
                modifier = Modifier
                    .width(2.dp)
                    .height(4.dp)
                    .align(Alignment.CenterStart)
                    .offset(x = (zeroPosition * 300).dp) # Approximate track width
            )
        }
    }
